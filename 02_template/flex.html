style 속성안내

flex 관련된 속성에 대한 간략 안내

.flex-wrap { 
	display:flex; 
	flex-direction:row; /* 하위 아이템 배치 방향 설정 */
	align-items:flex-start; /* flex-direction 에서 설정된 방향의 반대축에 대한 하위 아이템 배치 방향 설정 : baseline - 하위요소 텍스트 기준 정렬 */
	flex-wrap:nowrap; /* 배치 시 줄바뀜 설정 */
	/*flex-flow:row nowrap; - direction + wrap 동시 작성 */

	justify-content:space-evenly; /* 중심축 기준 아이템 정렬 및 배치 설정 */
	align-content:center; /* 반대축 기준 아이템 정렬 및 배치 설정 */ 
}
.item { 
	/* order:0; 각 아이템의 순서 변경 */
	flex-grow:1; /* 각 아이템이 중심축에서 차지하는 공간 비율 */
	flex-shrink:2; /* 각 아이템이 부모요소 공간에 따라 대응하는 수치 */
	flex-basis:10%; /* 각 아이템의 부모요소 대비 고정 너비(or 높이) 값 설정 */
	align-self:center; /* 각 아이템의 부모요소 대비 정렬 설정 */
}
