function orgSet(area){
	var wrap		= typeof area === "string" ? document.querySelector(area) : area,
		btns 		= wrap.querySelectorAll('button'),
		btnTrees 	= wrap.querySelectorAll('.btn-tree');
	
	Array.prototype.forEach.call(btnTrees, function(btn){
		var prt = btn.parentNode;
		btn.addEventListener('click', function(){
			prt.classList.contains('on') ? prt.classList.remove('on') : prt.classList.add('on');
		});
	});

	Array.prototype.forEach.call(btns, function(btn){
		if(!btn.previousElementSibling && !btn.classList.contains('btn-tree')) btn.classList.add('tree-last');
	});
}

orgSet('.organization-tree');
